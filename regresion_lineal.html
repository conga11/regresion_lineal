# Mostrar valores nulos
def df_nulos(df):
    df_nulos = pd.DataFrame({
        'Count of Null Values': df.isnull().sum(),
        'Null Values Percentage %': round(df.isnull().sum() / len(df) * 100, 2),
        'Data Type': df.dtypes,
        'First Value': df.iloc[0]
    })
    # pd.set_option('display.max_rows', None)
    return df_nulos

# Creamos una funcin para mostrar la correlacin entre dos variables

def correlacion(df, col_X, col_y):

    # Anlisis de correlacin
    print('\n\n')
    # Imprimir en formato de ttulo grande y en color rojo
    print('\033[1;31m' + f'\n\u001b[1;4mCorrelacin entre {col_X} y {col_y}:\u001b[0m')
    print('\n')

    # Filtramos el df para quedarnos con las columnas que nos interesan
    df = df[[col_X, col_y]]
    # Mostramos la correlacin entre las dos columnas
    print(df.corr())
    # Mostramos la grfica de dispersin
    # sns.scatterplot(x=col_X, y=col_y, color="blue", data=df)
    # plt.show()

    # modelo
    X = df[[col_X]]
    y = df[col_y]
    model = LinearRegression()
    model.fit(X, y)

    print('\n\nModel dict: ', model.__dict__)

    y_pred = model.predict(X)

    #Insertamos la columna de predicciones en el DataFrame
    df.insert(0, 'Predicciones', y_pred)

    #Visualizamos la grfica comparativa entre el total real y el total predecido
    sns.scatterplot(x=col_X, y=col_y, color="blue", data=df)
    sns.scatterplot(x=col_X, y='Predicciones', color="red", data=df)
    sns.lineplot(x=col_X, y='Predicciones', color="red", data=df)
    plt.show()

    # Mostramos el coeficiente de determinacin
    coef_det = model.score(X, y)
    print(f'Coeficente de determinacin: {coef_det}')

    # Mostramos el coeficiente de correlacin
    coef_corr = np.sqrt(coef_det)
    print(f'Coeficiente de correlacin: {coef_corr}')

    return coef_det, coef_corr
